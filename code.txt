        public static void emailsend(string txtfrom, string txtpassword, string txtto, string txttitle, string txtmessage)
        {
            try
            {
                MailMessage mail = new MailMessage();
                SmtpClient smtp = new SmtpClient("smtp.gmail.com");
                mail.From = new MailAddress(txtfrom);
                mail.To.Add(txtto);
                mail.Subject = txttitle;
                mail.Body = txtmessage;
                mail.IsBodyHtml = true;

                smtp.Port = 587;
                smtp.DeliveryMethod = SmtpDeliveryMethod.Network;
                smtp.UseDefaultCredentials = false;
                smtp.Credentials = new NetworkCredential(txtfrom, txtpassword);
                smtp.EnableSsl = true;
                smtp.Send(mail);

                Console.ForegroundColor = ConsoleColor.Green;
                Console.Write($"\nEmail Successfully Sent From: ");

                Console.ForegroundColor = ConsoleColor.Yellow;
                Console.Write($"{txtfrom}");
                Console.ResetColor();

                Console.ForegroundColor = ConsoleColor.Green;
                Console.Write(" To ");

                Console.ForegroundColor = ConsoleColor.Yellow;
                Console.Write($"{txtto}");
                Console.ResetColor();

                Console.WriteLine(" ");
            }
            catch (Exception ex)
            {
                Console.ForegroundColor = ConsoleColor.Yellow;
                Console.WriteLine($"\n{ex.Message}");
                Console.ResetColor();
            }
        }
